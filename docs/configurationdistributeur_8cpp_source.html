395  QString idProduit;
396  qDebug() << Q_FUNC_INFO << "requete" << requete;
397  baseDeDonnees->recuperer(requete, idProduit);
398  qDebug() << Q_FUNC_INFO << "idProduit" << idProduit;
399  requete = "UPDATE Bac SET idProduit = " + idProduit +
400  " WHERE idBac =" + QString::number(distributeur->getBac(i)->getIdBac()) + ";";
401  qDebug() << Q_FUNC_INFO << "requete" << requete;
402  baseDeDonnees->executer(requete);
403  int premierIndice = labelsPrix[i]->text().indexOf(QRegExp("\\d"));
404  int dernierIndice = labelsPrix[i]->text().lastIndexOf(QRegExp("\\d"));
405  QString prix = labelsPrix[i]->text().mid(premierIndice, dernierIndice - premierIndice + 1);
406  if(editionsNouveauPrix[i]->value() != 0.)
407  {
408  requete = "UPDATE Produit SET prix =" + prix + " WHERE idProduit = " + idProduit + ";";
409  qDebug() << Q_FUNC_INFO << "requete" << requete;
410  baseDeDonnees->executer(requete);
411  }
412  }
