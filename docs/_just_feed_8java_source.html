437  private ArrayList<Double> extraireRemplissages(@NonNull Message message)
438  {
439  ArrayList<Double> listeRemplissages = new ArrayList<Double>();
440 
441  try
442  {
443  JSONObject jsonMessage = new JSONObject(message.obj.toString());
444  JSONObject chargeUtile = jsonMessage.getJSONObject("uplink_message").getJSONObject("decoded_payload");
445  Iterator<?> clefs = chargeUtile.keys();
446  clefs.next();
447  while(clefs.hasNext())
448  {
449  String clef = (String) clefs.next();

451  }
452  Log.d(TAG, "remplissages : "+listeRemplissages);
453  }
454  catch (Exception e)
