255  public void handleMessage(@NonNull Message message)

260  switch(message.what)

269  Log.d(TAG, "[Handler] DECONNEXION_OK");
270  break;
271  case BaseDeDonnees.DECONNEXION_ERREUR:
272  Log.d(TAG, "[Handler] DECONNEXION_ERREUR");
273  break;
274  case BaseDeDonnees.REQUETE_SQL_OK:
275  Log.d(TAG, "[Handler] REQUETE_SQL_OK");
276  break;
277  case BaseDeDonnees.REQUETE_SQL_ERREUR:
278  Log.d(TAG, "[Handler] REQUETE_SQL_ERREUR");
279  break;
280  case BaseDeDonnees.REQUETE_SQL_SELECT:
281  Log.d(TAG, "[Handler] REQUETE_SQL_SELECT");
282  break;
283  case BaseDeDonnees.REQUETE_SQL_SELECT_DISTRIBUTEURS:
284  Log.d(TAG, "[Handler] REQUETE_SQL_SELECT_DISTRIBUTEURS");
285  afficherDistributeurs((ArrayList)message.obj);
286  break;
287  case BaseDeDonnees.REQUETE_SQL_SELECT_OPERATEURS:
288  Log.d(TAG, "[Handler] REQUETE_SQL_SELECT_OPERATEURS");
289  listeOperateurs = (ArrayList)message.obj;
290  invalidateOptionsMenu();
291  }
292  }
